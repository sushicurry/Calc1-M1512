		\begin{enumerate}
			\item[a.] Approximate the net area bounded by the graph of $f$ and $x$-axis on the interval using a midpoint Riemann sum with $n = 4$. 
			\vspace{8cm}
			\item[b.] Approximate the net area bounded by the graph of $f$ and $x$-axis on the interval using a right Riemann sum with $n = 8$.
			\newpage
			\item[c.] Write a formula for the right Riemann sum of $f$ on the interval with $n$ subintervals.  
			\vspace{8cm}
			\item[d.] Using the following formula $$\sum_{k = 1}^{n} k^2 = \frac{n (2n + 1)(n + 1)}{6}$$ evaluate the definite integral $\int_{0}^{2} f(x) \; dx$ by taking the limit as $n \to \infty$ of the formula you wrote in part c. 
		\end{enumerate}

		\item[2.] (12 pts) A softball is popped up vertically from the ground with a velocity of 30 m/s. 
		\begin{enumerate}
			\item[a.] Starting with the equation $a(t) = v'(t) = -9.8$ $\textnormal{m/s}^2$, find the equations for the velocity $v(t)$ and position $s(t)$. 
			\vspace{8cm}
			\item[b.] Find the time when the object reaches its highest point. What is the height? 
		\end{enumerate}
