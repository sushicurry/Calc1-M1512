\centerline{\textbf{\Large{Chain Rule Quiz}}}

\vspace{0.2in}
 

\begin{enumerate}

\item Let \[y = \frac{f(x)}{g(x)}\]. Show that \[y' = \frac{f'(x) g(x) - f(x) g'(x)}{[g(x)]^2}\]. \textbf{Do not use the quotient rule!}\textit{Hint:} Rewrite $y$ as a product and then differentiate. 

\end{enumerate}
